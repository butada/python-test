スクリプトを非表示にするボタン¶

In [24]:
from IPython.display import HTML

HTML("""
<button id="code-show-switch-btn"></button>

<script>
var code_show = true;

function switch_display_setting() {
    var switch_btn = $("#code-show-switch-btn");
    if (code_show) {
        $("div.input").hide();
        code_show = false;
        switch_btn.text("");
    }else {
        $("div.input").show();
        code_show = true;
        switch_btn.text("");
    }
}

$("#code-show-switch-btn").click(switch_display_setting);
</script>
""")
